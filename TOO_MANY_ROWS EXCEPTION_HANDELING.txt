CREATE TABLE STUDENT(SNO INT,
                     SNAME CHAR(20),
                     PER FLOAT);
INSERT INTO STUDENT VALUES(101,'SAI',70.88);
INSERT INTO STUDENT VALUES(102,'RAM',55.60);
INSERT INTO STUDENT VALUES(101,'KISHAN',70.88);

DECLARE
    U_SNAME CHAR(30);
    L INT;
BEGIN
    L:=&L;
    SELECT SNAME INTO U_SNAME FROM STUDENT WHERE SNO=L;

    DBMS_OUTPUT.PUT_LINE('STUDENT NAME='||U_SNAME);
EXCEPTION
    WHEN TOO_MANY_ROWS THEN
     DBMS_OUTPUT.PUT_LINE('ERROR TOO MANY ROWS');
END;
